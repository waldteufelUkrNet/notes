оголошення:
	display: flex;			/* для block-елементів */
	display: inline-flex;	/* для inline-елементів */
	
!!! В середині flex float не працює.

Усі елементи входять у контейнер за принципом LIFO (last in first out)

Контейнер має дві вісі: головну (main-axis) і поперечну (cross-axis).
Головна вісь може мати 4 напрямки: зверху вниз, знизу ввверх, зліва
напрово, справа наліво.
Поперечна вісь має тільки два напрямки: зверху вниз та зліва направо.
-------------------------------------------------------------------------

ВЛАСТИВОСТІ КОНТЕЙНЕРУ.
	
FLEX DIRECTION				напрям входження елементів у контейнер
	:row (default)			зліва направо
    :row-reverse			справа наліво
    :column					зверху вниз
    :column-reverse			знизу вверх

!!! flex-direction: row-reverse; = direction: rtl;

JUSTIFY-CONTENT				взаємне розташування елементів головної
							вісі та пробілів між ними
	:flex-start (default)	елементи на початку
	:flex-end				елементи в кінці
	:center					елементи по центру
	:space-between			між елементами рівні пробіли, крайні
							елементи прилягають до стінок контейнера
	:space-around			між елементами рівні пробіли, між стінками
							контейнера і крайніми елементами -
							половинні пробіли

ALIGN-ITEMS					взаємне розташування елементів поперечної
							вісі та пробілів між ними
	:flex-start 			елементи на початку
	:flex-end				елементи в кінці
	:center					елементи по центру
	:baseline				елементи вирівняні по базовій лінії (тексту
							в елементах)
	:stretch (default)		елементи розтягуються і займають увесь
							доступний простір (елементи не повинні мати
							явно вказаних розмірів)

FLEX-WRAP					дозвіл на обтікання
	:nowrap (default)		без обтікання
	:wrap					обтікання дозволене (надлишкові елементи
							стають знизу/справа)
	:wrap-rewerse			обтікання дозволене (надлишкові елементи
							стають зверху/зліва), змінюється напрям
							поперечної вісі

ALIGN-CONTENT				вирівнювання рядків (при flex-wrap: wrap;)
	:stretch (default)		рядки розтягуються і займають увесь
							доступний простір
	:flex-start (default)	рядки на початку
	:flex-end				рядки в кінці
	:center					елементи по центру
	:space-between			між рядками рівні пробіли, крайні рядки
							прилягають до стінок контейнера
	:space-around			між рядками рівні пробіли, між стінками
							контейнера і крайніми рядкамии - половинні
							пробіли
FLEX-FLOW
	A shorthand property for flex-direction and flex-wrap.

ВЛАСТИВОСТІ ЕЛЕМЕНТІВ.

ORDER			визначає порядок виводу едементів у flex-контейнер (від
				меншого до більшого) зачення за замовчуванням - 0
		
MARGIN			дозволяє керувати відступами.
	
	margin-right: auto;		весь вільний простір між елементами
							збереться справа від елементу.
	margin-left: auto;		весь вільний простір між елементами
							збереться зліва від елементу.
	margin: auto;			центрування елементу одразу по двох осях.
							Якщо елемент один, він стане по центру, якщо
							ні - вилізе з потоку.
							
!!! Якщо блочний елемент має явно вказані розміри, застосування одразу
margin-right: auto; та margin-left: auto; призведе до центрування
елементу.

ALIGN-SELF		дозволяє змінювати значення, прописані у властивості
				контейнеру align-items, для конкретного елементу.
	
	:auto 		бере батьківське значення align-items або stretch, якщо
				батька нема.
	:flex-start
	:flex-end
	:center
	:baseline
	:stretch

FLEX 			скорочення трьох властивостей (flex-grow, flex-shrink,
				flex-basis), значення за замовчуванням - 0 1 auto.
	
FLEX-GROW		визначає, як буде розподілений НАДЛИШКОВИЙ простір
				контейнеру, якщо сума розмірів елементів менша за
				розмір контейнеру. 
	.item1 { flex-grow: 2; }	/* елемент займе 50% вільного місця */
	.item2 { flex-grow: 1; }	/* елемент займе 25% вільного місця */
	.item3 { flex-grow: 1; }	/* елемент займе 25% вільного місця */
	
!!! Якщо елемент має flex-grow: 1;, а решта - 0, він займе увесь
вільний простір контейнеру. Під час зменшення розміру екрану буде
стискатися саме він до того часу, поки не віддасть увесь вільний
простір, потім елементи стискатимуться до розміру контенту рівномірно
або з коефіціентами, вказаними у flex-shrink).
	
FLEX-SHRINK		визначає, у скільки разів елемент буде меншим відносно
				інших елементів у випадку, якщо ширина контейнера менша
				за сумарну ширину елементів. Працює тільки при
				fex-wrap: nowrap;	
	
	.item1 {flex-shrink: 2;}	/* стискатиметься вдвічі швидше */
	.item2 {flex-shrink: 1;}
	
FLEX-BASIS		визначає початковий розмір елементу по ГОЛОВНІЙ вісі
				перед усіма наступними обчисленнями браузера. Може
				ПЕРЕвизначити ширину/висоту елементу (у залежності від
				flex-direction).
	 
	:auto 		автоматичний розмір відносно контенту.
	:значення	фіксоване значення у px, mm, pt або %.
